int red = 13;
int yellow = 12;
int green = 11;
int button = 2;

int state = 0;

unsigned long startTime = 0;
unsigned long duration = 0;

bool ped = false;

// таблица переходов
int table[3] = {1,2,0};

void setup() {

  pinMode(red, OUTPUT);
  pinMode(yellow, OUTPUT);
  pinMode(green, OUTPUT);
  pinMode(button, INPUT_PULLUP);

  Serial.begin(9600);

  state = 0;
  duration = 10000;
  startTime = millis();
}

void setLights() {

  digitalWrite(red, LOW);
  digitalWrite(yellow, LOW);
  digitalWrite(green, LOW);

  if (state == 0) digitalWrite(green, HIGH);
  if (state == 1) digitalWrite(yellow, HIGH);
  if (state == 2) digitalWrite(red, HIGH);
}

void loop() {

  setLights();

  if (digitalRead(button) == LOW) {
    ped = true;
    Serial.println("ped request");
    delay(200);
  }

  if (millis() - startTime > duration) {

    state = table[state];

    if (state == 0) duration = 10000;
    if (state == 1) duration = 3000;

    if (state == 2) {

      if (ped) {
        duration = 15000;
        ped = false;
      }
      else duration = 10000;
    }

    startTime = millis();

    Serial.print("state = ");
    Serial.println(state);
  }
}
