int red = 13;
int yellow = 12;
int green = 11;

int button = 2;

int state = 0; // 0=GREEN, 1=YELLOW, 2=RED

unsigned long startTime = 0;
unsigned long duration = 10000;

bool ped = false;

// для фиксации одного нажатия
int lastButtonState = HIGH;

void setup() {

  pinMode(red, OUTPUT);
  pinMode(yellow, OUTPUT);
  pinMode(green, OUTPUT);

  pinMode(button, INPUT);

  Serial.begin(9600);

  startTime = millis();
}

void loop() {

  // ---- чтение кнопки (реакция один раз) ----
  int buttonState = digitalRead(button);

  if (buttonState == LOW && lastButtonState == HIGH) {
    ped = true;
    Serial.println("Pedestrian button pressed");
  }

  lastButtonState = buttonState;

  // ---- автомат ----
  switch (state) {

    // GREEN
    case 0:

      digitalWrite(green, HIGH);
      digitalWrite(yellow, LOW);
      digitalWrite(red, LOW);

      if (millis() - startTime > duration) {

        state = 1;
        startTime = millis();
        duration = 3000;

        Serial.println("YELLOW");
      }

      break;

    // YELLOW
    case 1:

      digitalWrite(green, LOW);
      digitalWrite(yellow, HIGH);
      digitalWrite(red, LOW);

      if (millis() - startTime > duration) {

        state = 2;
        startTime = millis();

        if (ped) {
          duration = 15000; // длинный красный
          ped = false;
          Serial.println("RED with pedestrian");
        }
        else {
          duration = 10000;
          Serial.println("RED");
        }
      }

      break;

    // RED
    case 2:

      digitalWrite(green, LOW);
      digitalWrite(yellow, LOW);
      digitalWrite(red, HIGH);

      if (millis() - startTime > duration) {

        state = 0;
        startTime = millis();
        duration = 10000;

        Serial.println("GREEN");
      }

      break;
  }
}
